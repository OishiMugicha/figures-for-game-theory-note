};

% edges with labels
\draw[->] (root) -- (xcheck) node[midway,right, font=\large] {X checks};

\draw[->] (xcheck) -- (ybets) node[midway,right, font=\large] {Y bets};
\draw[->] (ybets) -- (xcall) node[midway,right, font=\large] {X calls};
